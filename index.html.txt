    <div id="dashboard">Loading stops...</div>

  <script>
    const ROUTES_CONFIG = [
      { route: "43X", direction: "O", target_stop: "KAM ON COURT", service_type: "1", stop_id: null },
      { route: "680", direction: "O", target_stop: "KAM CHUN COURT", service_type: "1", stop_id: null },
      { route: "A41P", direction: "O", target_stop: "PO TAI STREET", service_type: "1", stop_id: null }
    ];

    let countdown = 15;

    // Resolve Stop IDs once on initial load
    async function initStopIds() {
      for (let config of ROUTES_CONFIG) {
        const dir = config.direction === "I" ? "inbound" : "outbound";
        try {
          const res = await fetch(`https://data.etabus.gov.hk/v1/transport/kmb/route-stop/${config.route}/${dir}/${config.service_type}`);
          const data = await res.json();
          
          for (let item of data.data || []) {
            const stopInfoRes = await fetch(`https://data.etabus.gov.hk/v1/transport/kmb/stop/${item.stop}`);
            const stopInfo = await stopInfoRes.json();
            const stopName = (stopInfo.data?.name_en || "").toUpperCase();
            
            if (stopName.includes(config.target_stop.toUpperCase())) {
              config.stop_id = item.stop;
              config.resolved_name = stopInfo.data.name_en;
              break;
            }
          }
        } catch (e) {
          console.error(`Failed to lookup stop for ${config.route}:`, e);
        }
      }
      refreshDashboard();
    }

    async function fetchETA(stopId, route, targetDir) {
      if (!stopId) return [];
      try {
        const res = await fetch(`https://data.etabus.gov.hk/v1/transport/kmb/stop-eta/${stopId}`);
        const data = await res.json();
        return (data.data || []).filter(b => b.route === route && b.dir === targetDir && b.eta);
      } catch (e) {
        console.error(`ETA fetch error for ${route}:`, e);
        return [];
      }
    }

    async function refreshDashboard() {
      const container = document.getElementById('dashboard');
      let html = '';

      for (let config of ROUTES_CONFIG) {
        const etas = await fetchETA(config.stop_id, config.route, config.direction);
        const displayEtas = etas.slice(0, 3);

        html += `
          <div class="card">
            <div class="card-header">
              <span class="route-badge">${config.route}</span>
              <span class="stop-name">${config.resolved_name || config.target_stop}</span>
            </div>
            <ul class="eta-list">
        `;

        if (displayEtas.length === 0) {
          html += `<li class="eta-item" style="color: var(--subtext)">No upcoming departures</li>`;
        } else {
          displayEtas.forEach((bus, i) => {
            const etaTime = new Date(bus.eta);
            const now = new Date();
            const mins = Math.round((etaTime - now) / 60000);
            const timeStr = etaTime.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', hour12: false });

            let label = mins <= 0 ? 'Arriving Now' : `In ${mins} mins`;
            let isArriving = mins <= 0 ? 'arriving' : '';

            html += `
              <li class="eta-item">
                <span>Bus ${i + 1} (${timeStr})</span>
                <span class="eta-mins ${isArriving}">${label}</span>
              </li>
            `;
          });
        }

        html += `</ul></div>`;
      }

      container.innerHTML = html;
      document.getElementById('last-update').innerText = new Date().toLocaleTimeString([], { hour12: false });
      countdown = 15;
    }

    // Timer loop
    setInterval(() => {
      countdown--;
      if (countdown <= 0) {
        refreshDashboard();
      } else {
        document.getElementById('timer').innerText = countdown;
      }
    }, 1000);

    // Initialize
    initStopIds();
  </script>
